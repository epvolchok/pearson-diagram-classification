 60    def extract_observ_data(path: str, pattern: str = r'((?:r|t))swf-e_(\d{4})(\d{2})(\d{2})') -> Tuple[str, datetime.date]:
 61        """
 62        Extract observation type and date from the filename using regex.
 63
 64        Parameters
 65        ----------
 66        path : str
 67            File path or name.
 68        pattern : str
 69            Regular expression pattern to extract observation type and date.
 70
 71        Returns
 72        -------
 73        tuple
 74            (observation_type, datetime.date)
 75
 76        Raises
 77        ------
 78        ValueError
 79            If the pattern is not found.
 80        """
 81        search = re.search(pattern, path)
 82        if not search:
 83            logger.error(ValueError(f'Could not extract observation data from: {path}'))
 84            raise ValueError(f'Could not extract observation data from: {path}')
 85        obs_type = search.group(1)
 86        date = datetime.date(int(search.group(2)), int(search.group(3)), int(search.group(4)))
 87        return obs_type, date

 93        Prepare a directory for output, optionally clearing its contents.

263    @staticmethod
264    def split_into_two(df: pd.DataFrame, excluded_columns: Optional[list] = None) -> Tuple[pd.DataFrame, pd.DataFrame]:
265        """
266        Split a DataFrame into features and metadata columns.
267
268        Parameters
269        ----------
270        df : pandas.DataFrame
271            Input DataFrame.
272        excluded_columns : list of str, optional
273            Columns to exclude from the features set.
274
275        Returns
276        -------
277        tuple of pandas.DataFrame
278            (df_features, df_metadata)
279        """
280        if excluded_columns is None:
281            excluded_columns=['dataset_name', 'date', 'dist_to_sun[au]', 'SAMPLES_NUMBER', 
282                          'SAMPLING_RATE[kHz]', 'SAMPLE_LENGTH[ms]']
283        for possible_path in ['path', 'oldpath']:
284            if possible_path in df.columns:
285                if 'path' not in excluded_columns and 'oldpath' not in excluded_columns:
286                    excluded_columns = excluded_columns + [possible_path]
287                break
288
289        mask = [cols for cols in df.columns if cols not in excluded_columns]
290        df_features = df.loc[:, mask]
291        df_metadata = df.loc[:, excluded_columns]
292        return df_features, df_metadata

305    def get_log_filename(log_dir: str = 'logs', log_name: Optional[str] = None) -> str:
306        """
307        Determines a log file name.
308
309        Parameters
310        ----------
311        log_dir : str
312            Directory where log files are stored.
313        log_name : str or None
314            Custom log file name. If None, a new sequential name will be generated.
315
316        Returns
317        -------
318        str
319            Full path to the log file.
320        """
321        os.makedirs(log_dir, exist_ok=True)
322
323        if log_name:
324            if not log_name.endswith('.log'):
325                log_name += '.log'
326            return os.path.join(log_dir, log_name)
327
328        # Get existing log files
329        existing_logs = [f for f in os.listdir(log_dir) if f.startswith('log_') and f.endswith('.log')]
330        count = len(existing_logs)
331        next_num = count + 1
332
333        if count < 10:
334            name = f'log_{next_num:02d}.log'  # log_01.log, log_02.log, ...
335        else:
336            name = f'log_{next_num}.log'      # log_11.log, log_12.log, ...
337
338        return os.path.join(log_dir, name)

352            format='[%(asctime)s] [%(levelname)s] [%(name)s] %(message)s',
353            datefmt='%Y-%m-%d %H:%M:%S',
354            handlers=[
355                logging.FileHandler(log_file, mode='w'),
356                #logging.StreamHandler()
357            ]
358        )

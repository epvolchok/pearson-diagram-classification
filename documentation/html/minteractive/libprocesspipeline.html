 30    @staticmethod
 31    def get_features(input_imags: str, info_path: str, default_filename: str, name_pattern: str)-> ResNetFeatures:
 32        """
 33        Asks the user whether to extract new features or read from file.
 34
 35        Parameters
 36        ----------
 37        input_imags : str
 38            Subdirectory name with image data.
 39        info_path : str
 40            Path to the metadata file.
 41        default_filename : str
 42            Default base filename for feature data.

 65            elif choice == 'break':
 66                break

330        """
331        Executes the full default processing pipeline:
332        1. Feature filtering by variance
333        2. Dimensionality reduction with PCA and UMAP
334        3. Clustering with HDBSCAN
335        4. Visualization
336
337        Parameters
338        ----------
339        features : ResNetFeatures
340            Input feature database object.
341        default_filename : str
342            Base filename for saving outputs.
343        results_dir: str
344            Output directory.
345        """
346        message = 'The standard algorithm: \n' + \
347            '1. Filtration of ResNet features by variance, thershold=1e-5 \n' + \
348            '2. Preprocessing of the filtered features with \n' + \
349            '   - PCA(n_components=0.95, svd_solver=\'full\') \n' + \
350            '   - UMAP(n_components=20, min_dist=0.1, metric=\'cosine\') \n' + \
351            '3. Clusterization \n' + \
352            '- HDBSCAN(min_cluster_size=15, min_samples=5, metric=\'euclidean\') \n' + \
353            '4. Visualization with PCA+UMAP2D+HDBSCAN'
354        print(message)
355        logger.info('Standard algorithm launched')
356
357        features = ProcessingSteps.filtration(features)
358
359        message = f'Enter a file name to save the filtered data (or press "Enter" to use default name: {default_filename}) > '
360        PathManager.saving_database(features.database, default_filename, message, suf='_filtered')
361
362        processed = ProcessingSteps.run_preprocessing(features, 'PCA+UMAPND')
363
364        clusters = ProcessingSteps.run_clustering(processed, results_dir)
365
366        message = f'Enter a file name to save the clustered data (or press "Enter" to use default name: {default_filename}) > '
367        PathManager.saving_database(clusters.df, default_filename, message, suf='_clustered')
368
369        ProcessingSteps.evaluation_clustering_results(clusters, features, 'HDBSCAN')

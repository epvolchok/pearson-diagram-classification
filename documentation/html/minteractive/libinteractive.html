 16class InputManager:
 17    """
 18    A static utility class for handling interactive user input in the console.
 19
 20    Provides methods for:
 21    - Displaying welcome messages and prompts
 22    - Asking yes/no questions with validation
 23    - Parsing dictionaries from key=value input
 24    - Converting string input to appropriate Python types (bool, int, float, str)
 25    - Confirming or modifying regular expressions for dataset matching
 26
 27    This class is not meant to be instantiated.
 28    """
 29
 30    def __init__(self):
 31        ServiceFuncs.init_error(__class__.__name__)
 32    
 33    @staticmethod
 34    def input_wrapper(message: str) -> Optional[str]:
 35        """
 36        Prompts the user for input and handles KeyboardInterrupt gracefully.
 37
 38        Parameters
 39        ----------
 40        message : str
 41            Prompt message to display to the user.
 42
 43        Returns
 44        -------
 45        str or None
 46            User input string or None if interrupted.
 47        """
 48        try:
 49            user_input = input(message)
 50            return user_input
 51        except KeyboardInterrupt:
 52            logger.error('KeyboardInterrupt')
 53            print('\nInterrupted by user.')
 54            return
 55        
 56    @staticmethod
 57    def welcome_message() -> None:
 58        """
 59        Prints a welcome message with initial instructions for placing image data.
 60        """
 61
 62        message = 'Welcome to the script! \n' + \
 63        'First place your images in a separate folder (into "./images/") named according to the template: \n \
 64        "images_(your_specification)"'
 65        print(message)
 66        logger.info('Script started')

 68    @staticmethod
 69    def get_bool(prompt: str)-> bool: 
 70        """
 71        Prompts the user for a True/False response.
 72
 73        Parameters
 74        ----------
 75        prompt : str
 76            Prompt text for user input.
 77
 78        Returns
 79        -------
 80        bool
 81            Boolean interpretation of the user's input.
 82
 83        Raises
 84        ------
 85        KeyError
 86            If input is not 'true' or 'false' (case-insensitive).
 87        """
 88        while True:
 89            
 90            answ = InputManager.input_wrapper(prompt).lower()
 91            if answ:
 92                try:
 93                    return {'true': True, 'false': False}[answ]
 94                except KeyError as e:
 95                    print(f'Invalid input, please enter True or False! {e}')
 96                    logger.error(f'Invalid input, please enter True or False! {e}')
 97            else:
 98                return True

101    @staticmethod
102    def filter_mixed_freq(name_pattern: str) -> Tuple[bool, str]:
103        """
104        Asks the user whether to filter out mixed-frequency data.
105
106        Parameters
107        ----------
108        name_pattern : str
109            Default regular expression used to match dataset names.
110
111        Returns
112        -------
113        filter_mixed : bool
114            Whether to apply frequency filtering.
115        name_pattern : str
116            Possibly updated regex pattern provided by the user.
117        """
118
119        filter_mixed = InputManager.get_bool('Would you like to filter mixed frequencies? (True or False) ')
120        logger.info(f'Filter mixed frequencies: {filter_mixed}')
121        if filter_mixed:
122            print(f'Check the name pattern: {name_pattern}')
123            np_input = InputManager.input_wrapper('Change if it is needed or press "Enter": ').strip()
124            if np_input:
125                name_pattern = np_input
126        logger.info(f'Name pattern: {name_pattern}')
127        return filter_mixed, name_pattern
128    
129    @staticmethod
130    def input_dict(prompt: str = 'Enter pairs of key=value (empty string - end): ') -> dict:
131        """
132        Interactively builds a dictionary from user input of key=value pairs.
133
134        Parameters
135        ----------
136        prompt : str
137            Message shown before input begins.
138
139        Returns
140        -------
141        dict
142            Dictionary parsed from input.
143        """
144        print(prompt)
145        result = {}
146        while True:
147            line = InputManager.input_wrapper(' > ').strip()
148            if not line:
149                break
150            if '=' not in line:
151                print('Wrong format, Enter pairs of key=value')
152                continue
153            key, value = line.split('=', 1)
154            key = key.strip()
155            value = InputManager.parse_value(value.strip())
156            result[key] = value
157        return result
158    
159    @staticmethod
160    def parse_value(value: str) -> object:
161        """
162        Attempts to convert a string to a boolean, integer, float, or returns as string.
163
164        Parameters
165        ----------
166        value : str
167            Input string to parse.
168
169        Returns
170        -------
171        object
172            Parsed value (bool, int, float, or str).
173        """
174        
175        value = value.strip()
176        if value.lower() in {'true', 'yes'}:
177            return True
178        elif value.lower() in {'false', 'no'}:
179            return False
180        try:
181            return int(value)
182        except ValueError:
183            pass
184        try:
185            return float(value)
186        except ValueError:
187            pass
188        return value
189    
190    @staticmethod
191    def check_if_database(obj):
192        if isinstance(obj, ResNetFeatures):
193            df = obj.database
194        elif isinstance(obj, Clustering):
195            df = obj.df
196        elif isinstance(obj, pd.DataFrame):
197            df = obj
198        else:
199            logger.error(ValueError('(features) is non-known object'))
200            raise ValueError('(features) is non-known object')
201        return df

205class PathManager:
206    """
207    A static utility class for managing file paths, folders, and output saving.
208
209    Provides methods for:
210    - Prompting and validating image folder names
211    - Creating result and base directories
212    - Asking for custom or default paths
213    - Saving pandas DataFrames with user-defined filenames
214
215    This class is not meant to be instantiated.
216    """
217    def __init__(self):
218        ServiceFuncs.init_error(__class__.__name__)
219    
220    @staticmethod
221    def preparations() -> Tuple[str, str, str]:
222        """
223        Sets up required working directories and obtains the user's image folder name.
224
225        Returns
226        -------
227        input_imags : str
228            Name of the subdirectory in 'images/' containing image data.
229        default_filename : str
230            Base filename used for saving feature-related data.
231        results_dir : str
232            Directory for processed results.
233        """
234
235        base_dir_names = ['images', 'processed', 'figures', 'data', 'results', 'logs']
236        cwd = os.getcwd()
237        message = 'Use "processed" folder for processed images \n' + \
238        '"data" folder for info metadata \n' + \
239        '"figures" and "results" for graphic and data results, respectively.'
240        print(message)
241        
242        PathManager.create_base_dirs(*base_dir_names)
243
244        input_imags = PathManager.images_folder()
245        default_filename, results_dir = PathManager.create_results_dir(input_imags)
246        
247        return input_imags, default_filename, results_dir
248    
249    @staticmethod
250    def create_base_dirs(*args: str) -> None:
251        """
252        Ensures that a set of base directories exists (creates if missing).
253
254        Parameters
255        ----------
256        *args : str
257            Names of directories to ensure exist within the current working directory.
258        """
259        cwd = os.getcwd()
260        for dirname in args:
261            dirname = os.path.join(cwd, dirname)
262            ServiceFuncs.preparing_folder(dirname, clear=False)
263        logger.info('Base directories checked or created')
264    
265    @staticmethod
266    def images_folder() -> str:
267        """
268        Prompts the user to input a valid subdirectory name inside 'images/'.
269
270        Repeats until a valid directory is given.
271
272        Returns
273        -------
274        str
275            Valid image folder name.
276        """
277        cwd = os.getcwd()
278        input_imags = os.path.basename(InputManager.input_wrapper('Please enter the name of directory with images: '))
279        isdir = os.path.isdir(os.path.join(cwd, 'images', input_imags))
280
281        while not isdir:
282            print(f'Wrong folder name: {input_imags}. Try again.')
283            logger.debug(f'Wrong folder name: {input_imags}')
284            input_imags = os.path.basename(InputManager.input_wrapper('Please enter the name of directory with images: '))
285            isdir = os.path.isdir(os.path.join(cwd, 'images', input_imags))
286
287        return input_imags
288    
289    @staticmethod
290    def create_results_dir(input_imags: str) -> Tuple[str, str]:
291        """
292        Constructs paths for results and prompts whether to clear previous outputs.
293
294        Parameters
295        ----------
296        input_imags : str
297            Image folder name, used to derive output paths.
298
299        Returns
300        -------
301        default_filename : str
302            Base path for saving output data.
303        results_dir : str
304            Directory for saving processed files.
305        """
306        cwd = os.getcwd()
307        specification = ServiceFuncs.input_name(input_imags)
308        results_dir = os.path.join(cwd, 'processed', 'processed_'+specification)
309        default_filename = os.path.join(cwd, 'results', 'pearson_diagram_data_'+specification)
310        logger.info(f'Source images: {input_imags} Results: {results_dir, default_filename}')
311
312        clear = InputManager.get_bool(f'If folder {results_dir} already exists would you like to clear its contents? (True or False) ')
313        ServiceFuncs.preparing_folder(results_dir, clear=clear)
314        logger.info(f'Clearing results folder: {clear}')
315        return default_filename, results_dir

343        """
344        Prompts the user to save the DataFrame to a file with optional suffix.
345
346        Parameters
347        ----------
348        df : pandas.DataFrame

350        default_filename : str
351            Default file base name.
